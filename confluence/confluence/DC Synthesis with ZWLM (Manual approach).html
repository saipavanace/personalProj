############################################################
# Design compiler synthesis script - non-topographical mode

# To run it: First source the license file and invoke the tool
# source /engr/dev/tools/script/snps-eng.sh
# /engr/dev/tools/synopsys/syn_vQ-2019.12-SP5-2/syn/Q-2019.12-SP5-2/bin/dc_shell-t -64bit -f dc_synth_zwld.tcl -output_log_file dc.log
#############################################################

# Start Time
sh date
alias hi history

# Stop on errors!
set_app_var sh_continue_on_error false

# Library path
set library_path /engr/dev/tools/techno/library/tsmc/TCBN07/TSMCHOME/digital/Front_End/timing_power_noise/NLDM
set library_svt "tcbn07_bwph360l11p57pd_base_svt_090a"
set library_lvt "tcbn07_bwph360l11p57pd_base_lvt_090a"

set search_path [list "." $library_path]
set synthetic_library [list dw_foundation.sldb standard.sldb ]

set_app_var target_library "$library_path/$library_svt/tcbn07_bwph360l11p57pd_base_svtssgnp_0p675v_125c_rcworst_CCworst_T.db $library_path/$library_lvt/tcbn07_bwph360l11p57pd_base_lvtssgnp_0p675v_125c_rcworst_CCworst_T.db"

#set marco_library " "
set_app_var link_library [concat {*} $target_library $synthetic_library]

# RTL file pointers
set rtldir /scratch/smusini/test/kjeld/input/MemoryStudy
set filelist [ list \
$rtldir/prim_ff_ben.sv \
$rtldir/prim_ff_en.sv \
$rtldir/prim_latch.sv \
$rtldir/RF_nWben_mR_ports.sv \
$rtldir/RF_l0_p3_e128_w128.sv \
]

# create reports/outputs dirs
if {! [file exists reports]} {
file mkdir reports
}
if {! [file exists outputs]} {
file mkdir outputs
}

# create work direcory
set WORK_DIR ./work
if {! [file exists $WORK_DIR]} {
file mkdir $WORK_DIR
}

# set design lib
define_design_lib work -path ./work

# Hdlin options
set_app_var hdlin_preserve_sequential true

# read files
read_file -rtl -format sverilog -library work $filelist

# analyze files
analyze -format sverilog -library work $filelist

# set current_design
current_design RF_wrap
set top $current_design

# elaborate design
elaborate -library work $top

# save initial design
write_file -hierarchy -format ddc -output outputs/$top.precompile.ddc


# link design
link > reports/$top.link.rpt

# Fix Nets Driving Multiple Ports
set_fix_multiple_port_nets -all -buffer_constants [get_designs *]

# check_design
check_design -summary > reports/$top.check_design.rpt

# set operating conditions
set_operating_conditions -max ssgnp_0p675v_125c_rcworst_CCworst_T -library tcbn07_bwph360l11p57pd_base_lvtssgnp_0p675v_125c_rcworst_CCworst_T
set_wire_load_model -name ZeroWireload -library tcbn07_bwph360l11p57pd_base_lvtssgnp_0p675v_125c_rcworst_CCworst_T
set_wire_load_mode top

## Apply constraints
# clock freq = 1.6G; uncertainity = 15% + 5% added extra due to zero net delays
create_clock -period 0.625 -name CLK [get_ports CLK]
set_clock_uncertainty 0.125 [get_clocks CLK]

# IO delays 30:70 approach
set func_inputs [remove_from_collection [all_inputs] [all_clocks]]
set_input_delay -max 0.187 -clock [get_clocks CLK] $func_inputs
set_output_delay -max 0.437 -clock [get_clocks CLK] [all_outputs]

# Input transition time
set_input_transition 0.12 $func_inputs

# Output load
set_load 0.1 [all_outputs]

# transition
set_max_transition 0.15 $current_design

# Create path groups
puts "-I- Creating path groups"
set allMacros [all_macro_cells]
set allFlops [get_cells -quiet -hierarchical -filter "is_sequential==true && is_hierarchical==false && !is_macro_cell"]
if {[sizeof_collection $allMacros] > 0} {
group_path -name flop2macro -from $allFlops -to $allMacros
group_path -name macro2flop -from $allMacros -to $allFlops
}
group_path -name flop2flop -from $allFlops -to $allFlops
group_path -name io2flop -from [all_inputs] -to $allFlops
group_path -name flop2io -from $allFlops -to [all_outputs]
group_path -name feedthru -from [all_inputs] -to [all_outputs]

# Check and report timing before compile
check_timing -multiple_clock > reports/$top.check_timing.precompile.rpt

# Report timing loops
report_timing -loops > reports/$top.timing_loops.rpt

# uniquify the design
uniquify

# Ungroup starting from the second level of hierarchy
ungroup -all -start_level 2

## Compile command
compile_ultra -no_boundary_optimization -gate_clock

report_qor > reports/$top.qor.rpt

# Save compiled result of the top level
write_file -hierarchy -format ddc -output outputs/$top.gate_level.ddc
write_file -hierarchy -format verilog -output outputs/$top.gate_level.v

# Generating reports
report_qor > reports/$top.qor.rpt
report_timing -max_path 1000 -transition_time -nets -nosplit > reports/$top.max_timing.rpt
check_timing -multiple_clock > reports/$top.check_timing.rpt
report_area -hierarchy > reports/$top.area.rpt
report_constraint -significant_digits 3 > reports/$top.constraints.rpt
report_design > reports/$top.design.rpt

puts "-I- Finished Time: [date]"

exit
