Each ATU will have two parameters: nativeToCommon and CommonToNative.

Each will have an array of assignments that will look like below, 

nativeToCommon : [

                       {

                          "source" : "string", // The source of the assignment. Can be a concatenation of signals.

                          "destination" : "string", // The desination of the assignment. Cannot be a concatenation of signals.

                          "sChannel" : integer, // (Optional) Channel that the source comes from. This will be the channel with respect to the source. 

                          "dChannel" : integer // (Optional) Channel that the source comes from. This will be the channel with respect to the destination. 

                       }, ... 

                   ]


Restriction: If Source is made from a concatenation each signal must have a specified width. (e.g. id[3:0] and not just id)

Note: The Native Layer of the ATU will know how to interpret channel. (e.g. 0 - aw, 1 - ar),(e.g. 0 - b, 1 - r). The software would also have to be on the same page with this mapping. If there is only 1 channel like with Common Layer channel cannot be included or be set to 0.

Example

nativeToCommon : [ // aw/ar Channels to CTL

                       {

                          "source" : "{prot[3:0],1'b0,user[22:0]}", 

                          "destination" : "user", 

                          "sChannel" : 0 //(ar Channel),

                       },

                      {

                          "source" : "prot[3:0],user[23:0]", 

                          "destination" : "user", 

                          "sChannel" : 1 //(aw Channel)

                       },

                      {

                          "source" : "user[15:0]", 

                          "destination" : "dp_user", 

                          "sChannel" : 2 //(w Channel)

                       },

                      {

                          "source" : "axi_customer_protection[3:0]", 

                          "destination" : "dp_ib_prot", 

                          "sChannel" : 2 //(w Channel)

                       },

                       {

                          "source" : "user[24:23]", 

                          "destination" : "multi_label", 

                          "sChannel" : 0

                       }, 

                       {

                          "source" : "user[25:24]", 

                          "destination" : "multi_label", 

                          "sChannel" : 1

                       }

                   ]


commonToNative: [ // from CTL to r/b Channels

                       {

                          "source" : "{prot[3:0],1'b0,user[22:0]}", 

                          "destination" : "user", 

                          "dChannel" : 0 //(r Channel),

                       }, // to b user

                      {

                          "source" : "prot[3:0],user[23:0]", 

                          "destination" : "user", 

                          "dChannel" : 1 //(b Channel)

                       }  // to r user
