hw-lib/js/sym_csr_lib.js

Description

The function computes and returns the InterruptMap string as specified in HW-CTF Symphony Block Common Characteristics that is to be included in the "csrDescription" string for interrupt status registers in a csr. The inputs are the level at which the tree resides in the interrupt hierarchy, the maskMap used to specify the associated mask bit and register for each status bit, and the srcBlockIDMap that specifies the IDs of blocks that have interrupt status registers and are one level below in the interrupt hierarchy.

Usage

var interruptMap   = genInterruptMap( level, maskMap, { treeMap, srcBlockIDMap });

Return Value

The InterruptMap string. (See under section "Interrupt Bit descriptions" in HW-CTF Symphony Block Common Characteristics)

Parameters

  1. level : Type: string. Legal values: "TOP" (indicates top of interrupt hierarchy tree), "SUB" (all other levels in the tree).
  2. maskMap: Type: object. Key: Interrupt status bit offset, Value: An object with following keys: maskBit (bit offset of mask bit), maskRegister (name (string) of the mask register).
  3. treeMap: (Optional) Type: object. Key: Interrupt status bit offset, Value: A string that indicates which interrupt tree the interrupt belongs to. Legal values: "ERROR", "INFO". Default "ERROR".
  4. srcBlockIDMap: (Optional) Type: object. Key: interrupt status bit offset, Value: An array of strings each of which is a valid BlockID.

Examples:
var srcBlockIDMap         = { "1": [ 3 ], "2": [ 4 ], "5": [ "4.1", "4.2" ]  };

var treeMap                   = { "1": "INFO", "5": "ERROR" };

OR

var srcBlockIDMap         = [ [0], [1], [7] ];   // Because array is also an object

var maskMap                 = { "1": { maskBit: 1, maskRegister: "ABC" }, "2": { maskBit: 2, maskRegister: "XYZ" }, "5": { maskBit: 0, maskRegister: "PQR" }  };

OR

var maskMap                 = [ { maskBit: 1, maskRegister: "ABC" },  { maskBit: 2, maskRegister: "XYZ" }, { maskBit: 0, maskRegister: "PQR" }]; // Because array is also an object

var interruptMap   = genInterruptMap( "SUB", maskMap, srcBlockIDMap);
