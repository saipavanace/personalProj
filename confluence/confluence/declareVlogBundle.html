Path

hw-lib/js/lib_utils.js

Description

This function returns a string that has declarations for each of the class VlogSignal or class VlogArray instances each separated with a newline and optional indentation. 

Usage

const myVlogBundle                  = getVlogBundle( { type: string, tabs: a positive integer or zero }, vlogBundle );

Return Value

A string that has declarations for each value in the VlogBundle. 

Parameters

  1. type: Type: string. "wire" or "reg". Default: "wire".
  2. tabs: Type:  a positive integer or zero. Adds these many tabs at the beginning of each line except the first.
  3. vlogBundle: Type: object. This must be a VlogBundle.

Examples

// From new-style interfaces 
const instATP                              = new obj.userLib[ "InterfaceATP" ].                                            // Create an instance of interface of type ATP
const name                                 = "in_";
const signals                               = instATP.getSignalsBundle( { nVc: 1, wBus: 40, wProt: 0 } );       // Create a signals bundle object
const atpInterface                       = { name, signals };                                                                     // Old style interface
const inATP                                 = getVlogBundle( atpInterface );                                               // Convert old-style interface to a VlogBundle

\jsend
\=declareVlogBundle( { tabs: 1 }, inATP )=\                                                                                   // Prints declarations in multiple lines: line 0: wire in_valid; line1: wire in_ready; line2: wire [39 : 0 ] in_bus, line3: wire first, line4: wire last;
